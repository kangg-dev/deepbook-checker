// Địa chỉ Token $DEEP trên Sui const DEEP_TOKEN_ADDR = '0xdeeb61d8361160a3102fab5f70309a3eb3c72522e6ca00b561775878f088161'; async function updateTokenPrice() { const priceInput = document.getElementById('priceInput'); try { // Sử dụng API của DexScreener (Miễn phí, không cần Key, cập nhật nhanh) const response = await fetch(`https://api.dexscreener.com/latest/dex/tokens/${DEEP_TOKEN_ADDR}`); const data = await response.json(); if (data.pairs && data.pairs.length > 0) { const price = parseFloat(data.pairs[0].priceUsd); priceInput.value = price.toFixed(4); // Cập nhật vào ô nhập giá console.log(`[${new Date().toLocaleTimeString()}] Đã cập nhật giá $DEEP: $${price}`); // Nếu đang có dữ liệu trong bảng, tính toán lại luôn if (document.getElementById('resultArea').style.display === 'block') { calculateResultsOnly(); } } } catch (error) { console.error("Không thể lấy giá token:", error); } } // Hàm tính toán lại số liệu dựa trên giá mới mà không cần fetch lại API Point (để tiết kiệm tài nguyên) function calculateResultsOnly() { const rate = parseFloat(document.getElementById('rateInput').value) || 0; const price = parseFloat(document.getElementById('priceInput').value) || 0; let totalSumTokens = 0; let totalSumUsd = 0; // Lấy tất cả các dòng trong bảng để update lại cột USD const rows = document.querySelectorAll('#tableBody tr'); rows.forEach((row, index) => { const ptsTd = document.getElementById(`pts-${index}`); const tokTd = document.getElementById(`tok-${index}`); const usdTd = document.getElementById(`usd-${index}`); if (ptsTd && !ptsTd.classList.contains('loading') && !ptsTd.classList.contains('error')) { const points = parseFloat(ptsTd.innerText.replace(/,/g, '')) || 0; const tokens = points * rate; const usdValue = tokens * price; totalSumTokens += tokens; totalSumUsd += usdValue; tokTd.innerText = tokens.toLocaleString(undefined, {maximumFractionDigits: 2}); usdTd.innerText = usdValue.toLocaleString('en-US', {style: 'currency', currency: 'USD'}); } }); document.getElementById('totalTokens').innerText = totalSumTokens.toLocaleString(undefined, {maximumFractionDigits: 2}); document.getElementById('totalUsd').innerText = totalSumUsd.toLocaleString('en-US', {style: 'currency', currency: 'USD'}); }
